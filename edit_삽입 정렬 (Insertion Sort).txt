삽입 정렬(Insertion Sort)은 두 번째 원소부터 시작하여 앞쪽의 정렬된 부분과 비교하며 자신의 올바른 위치를 찾아 '삽입'하는 알고리즘으로, 데이터가 거의 정렬되어 있거나 크기가 작을 때 효율적입니다. 주로 실시간 데이터 정렬, 부분 정렬, 또는 다른 복잡한 정렬 알고리즘의 최적화된 서브루틴으로 사용됩니다.

def insertion_sort(arr):
    # 1번째 인덱스부터 시작
    for i in range(1, len(arr)):
        key = arr[i] # 현재 삽입할 값
        j = i - 1
        # 현재 값이 이전 값들보다 작으면 뒤로 밀기
        while j >= 0 and key < arr[j]:
            arr[j + 1] = arr[j]
            j -= 1
        # 적절한 위치에 key 삽입
        arr[j + 1] = key
    return arr

data = [64, 34, 25, 12, 22, 11, 90]
print(insertion_sort(data)) # 결과: [11, 12, 22, 25, 34, 64, 90]
